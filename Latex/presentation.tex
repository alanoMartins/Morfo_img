\begin{frame}
	\frametitle{Esqueleto}
	
	\begin{defn}
		Aumenta a espessura de um conjunto A
	\end{defn}
	
	\begin{eq}
		\begin{align*}
				S(A) = \bigcup^k_{k=0} \{ (A \circleddash kB) - [(A \circleddash kB) \circ B] \}
		\end{align*}
	\end{eq}
	
	Reconstrução de A:
	
	\begin{eq}
		\begin{align}
				S(A) = \bigcup^k_{k=0} \{ (A \circleddash kB) - [(A \circleddash kB) \circ B] \}
		\end{align}
	\end{eq}
	
	
	
\end{frame}
